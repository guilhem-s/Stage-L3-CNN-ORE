model = tensorflow.keras.models.Sequential()
model.add(tensorflow.keras.layers.BatchNormalization())
model.add(tensorflow.keras.layers.Conv2D(16, (3,3), activation = 'relu'))
model.add(tensorflow.keras.layers.MaxPool2D())
model.add(tensorflow.keras.layers.BatchNormalization())
model.add(tensorflow.keras.layers.Conv2D(64, (3,3), activation = 'relu'))
model.add(tensorflow.keras.layers.MaxPool2D())
model.add(tensorflow.keras.layers.BatchNormalization())
model.add(tensorflow.keras.layers.Conv2D(128, (3,3), activation = 'relu'))
model.add(tensorflow.keras.layers.Conv2D(512, (3,3), activation = 'relu'))
model.add(tensorflow.keras.layers.BatchNormalization())
model.add(tensorflow.keras.layers.GlobalAveragePooling2D())
model.add(tensorflow.keras.layers.Dense(200, activation = 'relu'))
model.add(tensorflow.keras.layers.BatchNormalization())
model.add(tensorflow.keras.layers.Dense(56, activation = 'sigmoid'))
model.build((None,300,300,1))
